val pid = PID(0.3, 0.0, 0.5)
val target = t

init {
    pid.setpoint = target.y
    pid.setOutputLimits(maxLinearAcceleration)
}

override fun move() {
    if (!reachedTarget(target)) {
        var acc = pid.getOutput(yPos)
        if (yPos > target.y) acc *= -1
        accelerateForward(acc)
    }
}
